```rust
//! Hotkey string parser.
//!
//! Parses hotkey strings like "Ctrl+Shift+F7" into (keysym, modifiers).

use crate::error::KeyflowError;

/// Parsed hotkey combination.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct HotkeyCombo {
    pub keysym: u32,
    pub modifiers: u16,
}

/// Shared callback wrapper (allows one callback to be registered for multiple keycodes).
pub(crate) type SharedCallback = std::sync::Arc<HotkeyCallback>;

// X11 modifier masks (from X.h)
pub const SHIFT_MASK: u16 = 1 << 0;    // ShiftMask = 1
pub const LOCK_MASK: u16 = 1 << 1;     // LockMask = 2 (Caps Lock)
pub const CONTROL_MASK: u16 = 1 << 2;  // ControlMask = 4
pub const MOD1_MASK: u16 = 1 << 3;     // Mod1Mask = 8 (usually Alt)
pub const MOD4_MASK: u16 = 1 << 6;     // Mod4Mask = 64 (usually Super)

/// Parse a modifier name to its X11 mask value.
fn parse_modifier(name: &str) -> Option<u16> {
    match name.to_lowercase().as_str() {
        "ctrl" | "control" => Some(CONTROL_MASK),
        "shift" => Some(SHIFT_MASK),
        "alt" => Some(MOD1_MASK),
        "super" | "win" | "meta" => Some(MOD4_MASK),
        _ => None,
    }
}
```

```rust
/// Map key name to X11 keysym value.
pub fn keysym_from_name(name: &str) -> Option<u32> {
    // Function keys F1-F24
    if let Some(rest) = name.strip_prefix('F') {
        if let Ok(n) = rest.parse::<u32>() {
            if (1..=24).contains(&n) {
                return Some(0xFFBE + n - 1); // XK_F1 = 0xFFBE
            }
        }
    }

    match name.to_uppercase().as_str() {
        // Letters A-Z (XK_a = 0x61, but we use uppercase convention)
        s if s.len() == 1 && s.chars().next().unwrap().is_ascii_alphabetic() => {
            let c = s.chars().next().unwrap().to_ascii_lowercase();
            Some(c as u32) // XK_a = 0x61, XK_b = 0x62, etc.
        }
        // Digits 0-9
        s if s.len() == 1 && s.chars().next().unwrap().is_ascii_digit() => {
            let c = s.chars().next().unwrap();
            Some(c as u32) // XK_0 = 0x30, XK_1 = 0x31, etc.
        }
        // Special keys
        "SPACE" => Some(0x0020),       // XK_space
        "TAB" => Some(0xFF09),         // XK_Tab
        "ESC" | "ESCAPE" => Some(0xFF1B), // XK_Escape
        "ENTER" | "RETURN" => Some(0xFF0D), // XK_Return
        "BACKSPACE" => Some(0xFF08),   // XK_BackSpace
        "DELETE" | "DEL" => Some(0xFFFF), // XK_Delete
        "INSERT" | "INS" => Some(0xFF63), // XK_Insert
        "HOME" => Some(0xFF50),        // XK_Home
        "END" => Some(0xFF57),         // XK_End
        "PAGEUP" | "PGUP" => Some(0xFF55), // XK_Page_Up
        "PAGEDOWN" | "PGDN" => Some(0xFF56), // XK_Page_Down
        "UP" => Some(0xFF52),          // XK_Up
        "DOWN" => Some(0xFF54),        // XK_Down
        "LEFT" => Some(0xFF51),        // XK_Left
        "RIGHT" => Some(0xFF53),       // XK_Right
        // Punctuation
        "MINUS" | "-" => Some(0x002d), // XK_minus
        "EQUAL" | "=" => Some(0x003d), // XK_equal
        "BRACKETLEFT" | "[" => Some(0x005b),
        "BRACKETRIGHT" | "]" => Some(0x005d),
        "BACKSLASH" | "\\" => Some(0x005c),
        "SEMICOLON" | ";" => Some(0x003b),
        "APOSTROPHE" | "'" => Some(0x0027),
        "GRAVE" | "`" => Some(0x0060),
        "COMMA" | "," => Some(0x002c),
        "PERIOD" | "." => Some(0x002e),
        "SLASH" | "/" => Some(0x002f),
        _ => None,
    }
}
```

```rust
/// Parse a hotkey string like "Ctrl+Shift+F7" into a HotkeyCombo.
pub fn parse_hotkey(hotkey: &str) -> Result<HotkeyCombo, KeyflowError> {
    let hotkey = hotkey.trim();
    if hotkey.is_empty() {
        return Err(KeyflowError::HotkeyRegistration {
            hotkey: hotkey.to_string(),
            reason: "empty hotkey string".to_string(),
        });
    }

    let parts: Vec<&str> = hotkey.split('+').map(|s| s.trim()).collect();
    let mut modifiers: u16 = 0;
    let mut keysym: Option<u32> = None;

    for part in &parts {
        if let Some(mod_mask) = parse_modifier(part) {
            modifiers |= mod_mask;
        } else if let Some(ks) = keysym_from_name(part) {
            if keysym.is_some() {
                return Err(KeyflowError::HotkeyRegistration {
                    hotkey: hotkey.to_string(),
                    reason: "multiple key names (only one non-modifier key allowed)".to_string(),
                });
            }
            keysym = Some(ks);
        } else {
            return Err(KeyflowError::HotkeyRegistration {
                hotkey: hotkey.to_string(),
                reason: format!("unknown key name: '{part}'"),
            });
        }
    }

    let keysym = keysym.ok_or_else(|| KeyflowError::HotkeyRegistration {
        hotkey: hotkey.to_string(),
        reason: "no key specified (only modifiers)".to_string(),
    })?;

    Ok(HotkeyCombo { keysym, modifiers })
}
```

```rust
// src/hotkey/linux.rs
//! Linux global hotkey implementation.
//!
//! Uses X11 XGrabKey for global hotkey registration.
//! Wayland support will be added later via libei.

use crate::error::KeyflowError;
use crate::hotkey::keys::{self, HotkeyCombo, CONTROL_MASK, SHIFT_MASK, MOD1_MASK, MOD4_MASK};
use crate::hotkey::{HotkeyCallback, HotkeyManager};
use std::collections::HashMap;
use std::sync::atomic::{AtomicBool, Ordering};
use std::sync::Arc;
use x11rb::connection::Connection;
use x11rb::protocol::xproto::*;
use x11rb::rust_connection::RustConnection;

/// Modifier masks for Num Lock and Caps Lock.
/// These are detected at startup via GetModifierMapping.
#[derive(Clone, Copy)]
struct LockModifiers {
    num_lock: u16,
    caps_lock: u16,
    scroll_lock: u16,
}

pub struct LinuxHotkeyManager {
    connection: RustConnection,
    root_window: Window,
    keymap: HashMap<u32, Vec<u8>>,                // keysym → keycodes
    callbacks: HashMap<(u8, u16), SharedCallback>,  // (keycode, base_modifiers) → callback
    registered_keys: Vec<(u8, u16)>,               // (keycode, full_modifiers) for cleanup
    lock_mods: LockModifiers,
    running: Arc<AtomicBool>,
}

impl LinuxHotkeyManager {
    pub fn new() -> Self {
        let (connection, screen_num) = RustConnection::connect(None)
            .expect("Failed to connect to X11 display. Is X running?");
        let screen = &connection.setup().roots[screen_num];
        let root_window = screen.root;

        // Build keysym → keycode mapping
        let keymap = build_keymap(&connection, screen);

        // Detect Num Lock / Caps Lock modifier masks
        let lock_mods = detect_lock_modifiers(&connection);

        Self {
            connection,
            root_window,
            keymap,
            callbacks: HashMap::new(),
            registered_keys: Vec::new(),
            lock_mods,
            running: Arc::new(AtomicBool::new(false)),
        }
    }
}

/// Build a mapping from keysym to keycode(s) using GetKeyboardMapping.
fn build_keymap(connection: &RustConnection, screen: &Screen) -> HashMap<u32, Vec<u8>> {
    let min_keycode = screen.min_keycode;
    let max_keycode = screen.max_keycode;
    let count = (max_keycode - min_keycode + 1) as u32;

    let reply = connection
        .get_keyboard_mapping(min_keycode, count)
        .unwrap()
        .reply()
        .unwrap();

    let keysyms_per_keycode = reply.keysyms_per_keycode as usize;
    let mut keymap: HashMap<u32, Vec<u8>> = HashMap::new();

    for (i, keycode) in (min_keycode..=max_keycode).enumerate() {
        let base = i * keysyms_per_keycode;
        for j in 0..keysyms_per_keycode {
            let keysym = reply.keysyms[base + j];
            if keysym != 0 {
                keymap.entry(keysym).or_default().push(keycode);
            }
        }
    }

    keymap
}

/// Detect modifier masks for Num Lock, Caps Lock, Scroll Lock.
fn detect_lock_modifiers(connection: &RustConnection) -> LockModifiers {
    let reply = connection.get_modifier_mapping().unwrap().reply().unwrap();
    let modmap = &reply.keycodes;
    let per_mod = reply.keycodes_per_modifier as usize;

    // Modifier indices: Shift=0, Lock=1, Control=2, Mod1=3, Mod2=4, Mod3=5, Mod4=6, Mod5=7
    // Num Lock is usually Mod2 or Mod4, Caps Lock is Lock (index 1)
    // We detect by checking which keycodes are mapped to each modifier

    let mut num_lock = 0u16;
    let mut caps_lock = 0u16;
    let mut scroll_lock = 0u16;

    // Get all keysyms for each keycode to identify lock keys
    let min_keycode = connection.setup().roots[0].min_keycode;
    let max_keycode = connection.setup().roots[0].max_keycode;
    let count = (max_keycode - min_keycode + 1) as u32;
    let km_reply = connection
        .get_keyboard_mapping(min_keycode, count)
        .unwrap()
        .reply()
        .unwrap();
    let kspkc = km_reply.keysyms_per_keycode as usize;

    // Helper: check if a keycode has a specific keysym
    let has_keysym = |keycode: u8, target: u32| -> bool {
        let idx = (keycode - min_keycode) as usize * kspkc;
        for j in 0..kspkc {
            if km_reply.keysyms[idx + j] == target {
                return true;
            }
        }
        false
    };

    // XK_Num_Lock = 0xFF7F, XK_Caps_Lock = 0xFFE5, XK_Scroll_Lock = 0xFF14
    for mod_idx in 0..8 {
        let mask: u16 = 1 << mod_idx;
        for k in 0..per_mod {
            let keycode = modmap[mod_idx * per_mod + k];
            if keycode == 0 {
                continue;
            }
            if has_keysym(keycode, 0xFF7F) {
                num_lock = mask;
            }
            if has_keysym(keycode, 0xFFE5) {
                caps_lock = mask;
            }
            if has_keysym(keycode, 0xFF14) {
                scroll_lock = mask;
            }
        }
    }

    LockModifiers {
        num_lock,
        caps_lock,
        scroll_lock,
    }
}
```

```rust
impl HotkeyManager for LinuxHotkeyManager {
    fn register(&mut self, hotkey: &str, callback: HotkeyCallback) -> Result<(), KeyflowError> {
        let combo = keys::parse_hotkey(hotkey)?;

        // Find keycode(s) for this keysym
        let keycodes = self.keymap.get(&combo.keysym).ok_or_else(|| {
            KeyflowError::HotkeyRegistration {
                hotkey: hotkey.to_string(),
                reason: format!("keysym 0x{:04X} not found on keyboard", combo.keysym),
            }
        })?;

        // Register for each keycode × lock combination
        let lock_masks = self.all_lock_masks();
        for &keycode in keycodes {
            for &lock_mask in &lock_masks {
                let full_modifiers = combo.modifiers | lock_mask;
                match self.connection.grab_key(
                    true,
                    self.root_window,
                    full_modifiers as u16,
                    keycode,
                    GrabMode::ASYNC,
                    GrabMode::ASYNC,
                ) {
                    Ok(_) => {
                        self.registered_keys.push((keycode, full_modifiers));
                    }
                    Err(e) => {
                        return Err(KeyflowError::HotkeyRegistration {
                            hotkey: hotkey.to_string(),
                            reason: format!("XGrabKey failed: {e:?}"),
                        });
                    }
                }
            }
        }

        // Store callback with base modifiers (no lock bits)
        // Use Arc so one callback can be shared across multiple keycodes
        let shared = std::sync::Arc::new(callback);
        for &keycode in keycodes {
            self.callbacks.insert((keycode, combo.modifiers), shared);
        }

        // Flush to ensure grab requests are sent
        self.connection.flush().map_err(|e| {
            KeyflowError::HotkeyRegistration {
                hotkey: hotkey.to_string(),
                reason: format!("flush failed: {e:?}"),
            }
        })?;

        log::info!("Registered hotkey: {hotkey} (keysym=0x{:04X}, mods=0x{:04X})", combo.keysym, combo.modifiers);
        Ok(())
    }
    // ... run, stop, running_flag 在下一个 task 实现
}

impl LinuxHotkeyManager {
    /// Get all combinations of lock modifier masks (for Num Lock / Caps Lock compatibility).
    fn all_lock_masks(&self) -> Vec<u16> {
        let lm = &self.lock_mods;
        let mut masks = vec![0u16];
        if lm.num_lock != 0 {
            let mut new = masks.clone();
            new.iter_mut().for_each(|m| *m |= lm.num_lock);
            masks.extend(new);
        }
        if lm.caps_lock != 0 {
            let mut new = masks.clone();
            new.iter_mut().for_each(|m| *m |= lm.caps_lock);
            masks.extend(new);
        }
        masks
    }
}
```

```rust
// 在 impl HotkeyManager for LinuxHotkeyManager 中添加

fn run(&self) -> Result<(), KeyflowError> {
    self.running.store(true, Ordering::SeqCst);
    log::info!("Hotkey manager started (Linux X11)");

    while self.running.load(Ordering::SeqCst) {
        // Non-blocking event check
        match self.connection.poll_for_event() {
            Ok(Some(event)) => {
                self.handle_event(event);
            }
            Ok(None) => {
                // No event, sleep briefly
                std::thread::sleep(std::time::Duration::from_millis(10));
            }
            Err(e) => {
                log::warn!("X11 event error: {e:?}");
                std::thread::sleep(std::time::Duration::from_millis(10));
            }
        }
    }

    log::info!("Hotkey manager stopped");
    Ok(())
}

fn stop(&self) {
    self.running.store(false, Ordering::SeqCst);
}

fn running_flag(&self) -> Arc<AtomicBool> {
    self.running.clone()
}
```

```rust
// 在 impl LinuxHotkeyManager 中添加

fn handle_event(&self, event: Event) {
    match event {
        Event::KeyPress(ev) => {
            let keycode = ev.detail;
            // Strip lock modifier bits for matching
            let state = ev.state & !(self.lock_mods.num_lock | self.lock_mods.caps_lock | self.lock_mods.scroll_lock);

            if let Some(callback) = self.callbacks.get(&(keycode, state)) {
                log::debug!("Hotkey triggered: keycode={keycode}, state=0x{state:04X}");
                callback();
            }
        }
        _ => {
            // Ignore other events
        }
    }
```
